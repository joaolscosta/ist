#include <stdio.h>

void junta(char s1[], char s2[])
{
    int i, j;
    for (i = 0; s1[i] != '\0'; i++)
    {
        continue;
    }
    for (j = 0; s2[j] != '\0'; j++)
    {
        s1[i + j] = s2[j];
    }
    s1[i + j] = '\0';
}

int main()
{
    char s1[] = "IAED: desde 1989.", s2[] = "IST: desde 1911.";
    junta(s1, s2);
    printf("%s", s1);
    return 0;
}

#include <stdio.h>

int comOcorrencia(int v[], int size)
{
    int i, nums[21] = {0}, res = 0;
    for (i = 0; i < size; i++)
    {
        nums[v[i]]++;
    }
    for (i = 0; i < 21; i++)
    {
        if (nums[i] != 0)
        {
            res++;
        }
    }
    return res;
}

int main()
{
    int v[] = {1, 5, 2, 3, 10, 15, 20, 2, 5, 10, 12, 15};
    printf("%d", comOcorrencia(v, 12));
    return 0;
}

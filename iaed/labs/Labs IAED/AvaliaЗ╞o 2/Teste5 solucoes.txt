#include <stdio.h>

int seqCrescente(int v[], int size)
{
    int i = 0, j, max = 0, cur = 1;
    for (i = 0; i < size; i++)
    {
        for (j = i; j < size - 1; j++)
        {
            if (v[j + 1] > v[j])
            {
                cur++;
            }
            else
            {
                if (cur > max)
                {
                    max = cur;
                }
                cur = 1;
                break;
            }
        }
    }
    return max;
}

int main()
{
    int v[] = {1, 2, 1, 10, 12, 18, 7, 8, 9};
    printf("%d", seqCrescente(v, 9));
    return 0;
}

#include <stdio.h>

int semOcorrencia(int v[], int size)
{
    int i, nums[21] = {0}, res = 0;
    for (i = 0; i < size; i++)
    {
        nums[v[i]]++;
    }
    for (i = 0; i < 21; i++)
    {
        if (nums[i] == 0)
        {
            res++;
        }
    }
    return res;
}

int main()
{
    int v[] = {1, 5, 2, 3, 10, 15, 20, 2, 5, 10, 12, 15};
    printf("%d", semOcorrencia(v, 12));
    return 0;
}

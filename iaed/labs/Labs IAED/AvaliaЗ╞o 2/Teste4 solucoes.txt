#include <stdio.h>

int produtoInterno(int v1[], int v2[], int size)
{
    int i, res = 0;

    for (i = 0; i < size; i++)
    {
        res += (v1[i] * v2[i]);
    }
    return res;
}

int main()
{
    int v1[] = {0, 2, 4}, v2[] = {1, 3, 5};
    printf("%d", produtoInterno(v1, v2, 3));
    return 0;
}

#include <stdio.h>

int maiusculaFrequente(char s[])
{
    int i, v[130] = {0}, max = 0;
    char res;
    for (i = 0; s[i] != '\0'; i++)
    {
        if (s[i] >= 'A' && s[i] <= 'Z')
        {
            v[(int)s[i]]++;
        }
    }
    for (i = 0; i < 130; i++)
    {
        if (v[i] > max)
        {
            max = v[i];
            res = i;
        }
    }
    return (char)res;
}

int main()
{
    char s[] = "LEIC e IST: juntos desde 1989.";
    printf("%c", maiusculaFrequente(s));
    return 0;
}

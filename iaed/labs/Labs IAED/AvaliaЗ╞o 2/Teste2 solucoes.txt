#include <stdio.h>

int contaRepeticoes(int v[], int size)
{
    int i, res = 0, nums[10] = {0};

    for (i = 0; i < size; i++)
    {
        nums[v[i]]++;
    }
    for (i = 0; i < 10; i++)
    {
        if (nums[i] > 1)
        {
            res += (nums[i] - 1);
        }
    }
    return res;
}

int main()
{
    int v[8] = {3, 6, 4, 6, 6, 4, 5, 8};
    printf("%d", contaRepeticoes(v, 8));
    return 0;
}

#include <stdio.h>

int minusculasComuns(char s1[], char s2[])
{
    int res = 0, i, j, v[130] = {0};
    for (i = 0; s1[i] != '\0'; i++)
    {
        if (s1[i] >= 'a' && s1[i] <= 'z')
        {
            for (j = 0; s2[j] != '\0'; j++)
            {
                if (s1[i] == s2[j])
                {
                    if (v[(int)s1[i]] == 0)
                    {
                        v[(int)s1[i]] = 1;
                        res++;
                    }
                }
            }
        }
    }
    return res;
}

int main()
{
    char s1[] = "LEIC: excelente desde 1989.", s2[] = "IST-21: a projectar o futuro desde 1911.";
    printf("%d", minusculasComuns(s1, s2));
    return 0;
}
